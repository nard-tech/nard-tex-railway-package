  \subsection{JR北海道}

  \begin{itemize}
    \item \verb|\JRHokkaido| - \JRHokkaido
    \item \verb|\ArticleJRHokkaido{条文番号}|
      \begin{itembox}[l]{コード}
        \begin{verbatim}
\ArticleJRHokkaido{1}
\begin{quote}
  これはJR北海道の第1条の内容です。
\end{quote}\end{verbatim}
      \end{itembox}
      \begin{itembox}[l]{表示内容}
        \ArticleJRHokkaido{1}
        \begin{quote}
          これはJR北海道の第1条の内容です。
        \end{quote}
      \end{itembox}
    \item \verb|\ArticleSubJRHokkaido{条文番号}{枝番}|
